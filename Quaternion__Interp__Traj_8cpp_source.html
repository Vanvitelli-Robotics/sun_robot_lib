160  /*
161  initialize the trajectory
162  */
163  void Quaternion_Linear_Traj_Gen::initialize(const Matrix<4,4>& b_T_init){
164 
165  _b_quat_init = UnitQuaternion(b_T_init);
166 
167  switch (_mode)
168  {
169  case Robot_Motion_Mode::REL_TOOL : {
170 
171  //In this case the _init_angvec_desired is already in the correct frame
172  //_b_rotax is the rotation axis of the _init_angvec_desired rotated in the {base} frame
173 
174  _b_rotax = b_T_init.slice<0,0,3,3>() * _init_angvec_desired.getVec();
175 
176  break;
177  }
178  case Robot_Motion_Mode::REL_BASE : {
179 
180  //In this case the _init_angvec_desired has the axis in the {base} frame
181  // but the angle is already a delta_angle
182 
183  //_b_rotax is the axis used as input in the constructor
184  _b_rotax = _init_angvec_desired.getVec();
185 
186  //The correct axis of _init_angvec_desired is _b_rotax rotated in the {initial} frame
187  _init_angvec_desired.setVec( b_T_init.slice<0,0,3,3>().T() * _b_rotax );
188 
189  break;
190  }
191  case Robot_Motion_Mode::ABS_BASE : {
192 
193  //In this case the _init_angvec_desired has the axis in the {base} frame
194  // and the angle is a absolute angle w.r.t. base frame
195 
196  //_init_angvec_desired at the left of this line has to been read as _b_angvec_desired
197  UnitQuaternion init_quat_desired = inv(_b_quat_init) * UnitQuaternion(_init_angvec_desired);
198  //Here the correct _init_angvec_desired is computed
199  _init_angvec_desired = init_quat_desired.toangvec();
200 
201  //Now it is necessary reconstruct the _quintic_angle generator using the new delta_angle
202  /*
203  _quintic_angle = Quintic_Poly_Gen(
204  _initial_time,
205  _final_time,
206  0.0,
207  _init_angvec_desired.getAng(),
208  _wi,
209  _wf,
210  _dwi,
211  _dwf
212  );*/
213  _quintic_angle.setFinalPosition(_init_angvec_desired.getAng());
214 
215  //Now the situation is the same of the case Robot_Motion_Mode::REL_TOOL
216  _b_rotax = b_T_init.slice<0,0,3,3>() * _init_angvec_desired.getVec();
217 
218  break;
219  }
220 
221  default:{
222  cout << TRAJ_ERROR_COLOR << "[Quaternion_Linear_Traj_Gen] Error in initialize() - invalid MODE" CRESET << endl;
223  exit(-1);
224  }
225  }
