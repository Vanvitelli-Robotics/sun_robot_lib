82  Matrix<4,4> RobotLink::A_internal( double theta, double d ) const{
83 
84  double sa = sin(_alpha);
85  double ca = cos(_alpha);
86 
87  double st = sin(theta);
88  double ct = cos(theta);
89 
90  //standard DH
91  return Data(
92  ct, -st*ca, st*sa, _a*ct,
93  st, ct*ca, -ct*sa, _a*st,
94  0.0, sa, ca, d,
95  0.0, 0.0, 0.0, 1.0
96  );
97 
98  }
