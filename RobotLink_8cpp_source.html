75 Matrix<4, 4> RobotLink::A_internal(double theta, double d) const
76 {
77  double sa = sin(_alpha);
78  double ca = cos(_alpha);
79 
80  double st = sin(theta);
81  double ct = cos(theta);
82 
83  // standard DH
84  return Data(ct, -st * ca, st * sa, _a * ct, st, ct * ca, -ct * sa, _a * st, 0.0, sa, ca, d, 0.0, 0.0, 0.0, 1.0);
