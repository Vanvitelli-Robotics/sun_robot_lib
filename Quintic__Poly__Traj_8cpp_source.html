176 /*
177  Update poly coefficients
178 */
179 void Quintic_Poly_Traj::updateCoefficients(){
180  //This is the effective total time of motion
181  double t = getDuration();
182 
183  if(t < 10.0*std::numeric_limits<double>::epsilon()){
184  cout << TRAJ_WARN_COLOR "[Quintic_Poly_Traj] WARNING: duration is zero... I will fix this..." CRESET << endl;
185  t = QUINTIC_POLY_EPS_TIME;
186  }
187 
188  //These coefficients are known
189  _poly_coeff[5] = _pi;
190  _poly_coeff[4] = _vi;
191  _poly_coeff[3] = _aci/2.0;
192 
193  //Prepare pow
194  double t_2 = pow(t,2);
195  double t_3 = pow(t,3);
196  double t_4 = pow(t,4);
197  double t_5 = pow(t,5);
198 
199  //Construct inverse of A ( A*poly_coeff = B )
200  Matrix<3,3> A_inv = Data(
201  6.0/t_5, -3.0/t_4, 1.0/(2.0*t_3),
202  -15.0/t_4, 7.0/t_3, -1.0/t_2,
203  10.0/t_3, -4.0/t_2, 1.0/(2.0*t)
204  );
205  //Construct B
206  Vector<3> B = makeVector(
207  _pf - _pi - _vi*t - (_aci/2.0)*t_2,
208  _vf - _vi - _aci*t,
209  _acf - _aci
210  );
211 
212  //Calculate poly coeff
213  _poly_coeff.slice<0,3>() = A_inv*B;
214 
215  //calculate coeff of dp and ddp as polydiff
216  _vel_poly_coeff = polydiff(_poly_coeff);
217  _acc_poly_coeff = polydiff(_vel_poly_coeff);
218 
219 }
