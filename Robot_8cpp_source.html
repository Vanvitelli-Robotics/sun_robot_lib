539  /*
540  Internal fkine
541  This function compute the fkine to joint "n_joint" given the last transformation to joint n_joint-1
542  - q_DH_j is the joint position of the i-th link
543  - b_T_j_1 is the transformation of the link j-1 w.r.t base frame
544  */
545  Matrix<4,4> Robot::fkine_internal(const double& q_DH_j, const Matrix<4,4>& b_T_j_1, int n_joint ) const{
546  return ( b_T_j_1 * _links[n_joint]->A(q_DH_j) );
547  }

549  /*
550  fkine to n_joint-th link

553  */
554  Matrix<4,4> Robot::fkine( const Vector<>& q_DH, int n_joint, const Matrix<4,4>& j_T_f ) const{
555  return (fkine( q_DH, n_joint ) * j_T_f);

558  /*
559  fkine to n_joint-th link
560  if n_joint = NUM_JOINT+1 then the result is b_T_e
561  */
562  Matrix<4,4> Robot::fkine( const Vector<>& q_DH, int n_joint ) const{
563 
564  //Start from frame 0
565  Matrix<4,4> b_T_j = _b_T_0;
566 
567  //check if the final frame is {end-effector}
568  bool ee = false;
569  if( n_joint == (getNumJoints()+1) ){
570  n_joint --;
571  ee = true;
572  }
573 
574  for( int i = 0; i < n_joint; i++ ){
575  b_T_j = fkine_internal( q_DH[i], b_T_j, i );
576  }
577 
578  //if the final frame is the {end-effector} then add it
579  if(ee){
580  b_T_j = b_T_j * _n_T_e;
581  }
582 
583  return b_T_j;
584 
585  }

639  /*
640  Internal computation of the position part of the jacobian in the frame {f}
641  The input is a vector of all transformation the considered joints i.e.

643  */
644  Matrix<3,Dynamic> Robot::jacob_p_internal( const vector<Matrix<4,4>>& all_T ) const{
645 
646  int numQ = all_T.size()-1;
647 
648  Matrix<3,Dynamic> Jp = Zeros(3,numQ);
649 
650  Vector<3> p_e = all_T.back().T()[3].slice<0,3>();
651 
652  for( int i = 0; i<numQ ; i++ ){
653 
654  Vector<3> z_i_1 = all_T[i].T()[2].slice<0,3>();
655 
656  switch( _links[i]->type() ){
657 
658  case 'p': //Prismatic
659  {
660 
661  Jp.T()[i] = z_i_1;
662 
663  break;
664 
665  }
666 
667  case 'r': //Revolute
668  {
669 
670  Vector<3> p_i_1 = all_T[i].T()[3].slice<0,3>();
671 
672  Jp.T()[i] = z_i_1 ^ ( p_e - p_i_1 );
673 
674  break;
675 
676  }
677 
678  default:{
679  cout << ROBOT_ERROR_COLOR "[Robot] Error in jacob_p_internal( const vector<Matrix<4,4>>& all_T ): invalid _links["<<i<<"].type()="<<_links[i]->type() << ROBOT_CRESET << endl;
680  exit(-1);
681  }
682 
683  }
684 
685  }
686 
687  return Jp;
