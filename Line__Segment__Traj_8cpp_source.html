147 /*
148  Change the reference frame of the trajectory
149  Apply an homogeneous transfrmation matrix to the trajectory
150  new_T_curr is the homog transf matrix of the current frame w.r.t. the new frame
151 */
152 void Line_Segment_Traj::changeFrame( const Matrix<4,4>& new_T_curr ) {
153  Vector<4> tmp_v;
154  tmp_v[3] = 1.0;
155 
156  tmp_v.slice<0,3>() = _pi;
157  _pi = (new_T_curr * tmp_v).slice<0,3>();
158 
159  tmp_v.slice<0,3>() = _pf;
160  _pf = (new_T_curr * tmp_v).slice<0,3>();
161 }
